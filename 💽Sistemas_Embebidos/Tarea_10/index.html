    El objetivo es por medio de una simulaci√≥n en WOKWI o en f√≠sico leer un sensor RTC (Real Time Clock) y un aceler√≥metro.Todo esto se debe de realizar utilizando el m√©todo de lectura y escritura de datos I2C.

    Real Time Clock

    #include "pico/stdlib.h"
#include "hardware/i2c.h"
#include <stdio.h>

#define I2C_PORT i2c0
#define RTC_SDA 4
#define RTC_SCL 5
#define RTC_ADDR 0x68  // Direcci√≥n del DS1307 o DS3231

// Convierte de Decimal codificado en binario (BCD) a decimal
uint8_t bcd_to_dec(uint8_t val) {
    return ((val / 16) * 10) + (val % 16);
}

// Funci√≥n para leer la hora del RTC
void rtc_read_time() {
    uint8_t buffer[3];
    uint8_t reg = 0x00;  // Registro inicial (segundos)

    // Escribir direcci√≥n inicial de lectura
    i2c_write_blocking(I2C_PORT, RTC_ADDR, &reg, 1, true);//No env√≠a STOP, deja el bus abierto para leer inmediatamente despu√©s.
    // Leer segundos, minutos y horas
    i2c_read_blocking(I2C_PORT, RTC_ADDR, buffer, 3, false);// Ya termin√© de enviar los datos, cierro la comunicaci√≥n.

    uint8_t seconds = bcd_to_dec(buffer[0] & 0x7F);//Borra el bit 7, ya que ese no contiene informaci√≥n de los segundos
    uint8_t minutes = bcd_to_dec(buffer[1]);
    uint8_t hours = bcd_to_dec(buffer[2] & 0x3F);

    printf("Hora actual: %02d:%02d:%02d\n", hours, minutes, seconds);
}

int main() {
    stdio_init_all();

    // Inicializar I2C en los pines 0 (SDA) y 1 (SCL)
    i2c_init(I2C_PORT, 100 * 1000);// se comunican a 100KHZ (velocidad est√°ndar)
    gpio_set_function(RTC_SDA, GPIO_FUNC_I2C);
    gpio_set_function(RTC_SCL, GPIO_FUNC_I2C);
    gpio_pull_up(RTC_SDA);// Activamos las resistencias de forma que est√© en 1 constante
    gpio_pull_up(RTC_SCL);//de esta forma solo necesitamos bajara a GND no generar voltaje

    sleep_ms(500);
    printf("Lectura de hora desde RTC iniciada...\n");

    while (true) {// Permite que mande la lectura de tiempo cada 1 segundos
        rtc_read_time();
        sleep_ms(1000);
    }
}

    Aceler√≥metro

    #include <stdio.h>
#include "pico/stdlib.h"
#include "hardware/i2c.h"

#define MPU6050_ADDR 0x68
#define SDA_PIN 8
#define SCL_PIN 9

int main() {
    stdio_init_all();
    i2c_init(i2c0, 400 * 1000);
    gpio_set_function(SDA_PIN, GPIO_FUNC_I2C);
    gpio_set_function(SCL_PIN, GPIO_FUNC_I2C);
    gpio_pull_up(SDA_PIN);
    gpio_pull_up(SCL_PIN);

    sleep_ms(1000);
    printf("MPU6050 b√°sico iniciado...\n");

    // Despertar el sensor (quitar sleep mode)
    uint8_t wake[2] = {0x6B, 0x00};
    i2c_write_blocking(i2c0, MPU6050_ADDR, wake, 2, false);

    uint8_t reg = 0x3B;
    uint8_t data[14];
    int16_t ax, ay, az, gx, gy, gz;

    while (true) {
        // Leer 14 bytes desde el registro 0x3B (aceler√≥metro y giroscopio)
        i2c_write_blocking(i2c0, MPU6050_ADDR, &reg, 1, true);
        i2c_read_blocking(i2c0, MPU6050_ADDR, data, 14, false);

        ax = (data[0] << 8) | data[1];
        ay = (data[2] << 8) | data[3];
        az = (data[4] << 8) | data[5];
        gx = (data[8] << 8) | data[9];
        gy = (data[10] << 8) | data[11];
        gz = (data[12] << 8) | data[13];

        printf("A[X:%6d Y:%6d Z:%6d] | G[X:%6d Y:%6d Z:%6d]\n",
               ax, ay, az, gx, gy, gz);

        sleep_ms(500);
    }
}
